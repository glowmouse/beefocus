118 void FocuserState::processCommand( CommandParser::CommandPacket cp )
119 {
120  DebugInterface& log = *debugLog;
121 
122  // Status was originally before abort. TODO, refactor this mess.
123 
124  if ( cp.command == CommandParser::Command::Status )
125  {
126  State state = top().state;
127  int arg0 = top().arg0;
128 
129  log << "Processing pstatus request\n";
130  *net << "Position: " << focuserPosition << "\n";
131  *net << "State: " << stateNames.at(state) << " " << arg0 << "\n";
132  return;
133  }
134 
135  if ( cp.command == CommandParser::Command::PStatus )
136  {
137  log << "Processing pstatus request\n";
138  *net << "Position: " << focuserPosition << "\n";
139  return;
140  }
141 
142  if ( cp.command == CommandParser::Command::SStatus )
143  {
144  State state = top().state;
145  int arg0 = top().arg0;
146 
147  log << "Processing sstatus request\n";
148  *net << "State: " << stateNames.at(state) << " " << arg0 << "\n";
149  return;
150  }
151 
152  if ( cp.command == CommandParser::Command::Abort ) {
153  // Abort command, unroll the state stack and start accepting commands.
154  hard_reset_state( State::ACCEPT_COMMANDS, 0 );
155  return;
156  }
157 
158  if ( cp.command == CommandParser::Command::Home ) {
159  hard_reset_state( State::ACCEPT_COMMANDS, 0 );
160  pushState( State::STOP_AT_HOME );
161  return;
162  }
163 
164  if ( cp.command == CommandParser::Command::ABSPos ) {
165  pushState( State::MOVING, cp.optionalArg );
166  int new_position = cp.optionalArg;
167 
168  if ( new_position < focuserPosition )
169  {
170  int backtrack = new_position - 500;
171  backtrack = backtrack < 0 ? 0 : backtrack;
172  pushState( State::MOVING, backtrack );
173  }
174  }
175 
176  if ( cp.command == CommandParser::Command::Sleep ) {
177  pushState( State::LOW_POWER, 0 );
178  }
179  if ( cp.command == CommandParser::Command::Wake )
180  {
181  hard_reset_state( State::ACCEPT_COMMANDS, 0 );
182  pushState( State::AWAKEN );
183  }
